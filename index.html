1. Find the missing number in the array

You are given an array of positive numbers from 1 to n, such that all numbers from 1 to n are present except one number x. You have to find x. The input array is not sorted. Look at the below array and give it a try before checking the solution.

let find_missing = function(input) {
// calculate sum of all integers
// in input list

let actual_sum = 0;
for (let i in input) {
actual_sum += input[i];
}
// There is exactly 1 number missing
let n = input.length + 1;
let sum_of_n = Math.floor((n * (n + 1)) / 2);
return sum_of_n - actual_sum;
};

let random_array = [1,2,4];
let actual_missing = find_missing(random_array);
console.log("actual missing is " + actual_missing);
