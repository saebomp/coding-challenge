let find_missing = function(input) {
// calculate sum of all integers
// in input list

let actual_sum = 0;

actual_sum += input[i];
}
// There is exactly 1 number missing
let n = input.length + 1;

2. Determine if the sum of two integers is equal to the given value (Array안에 두 숫자의 합인 숫자가 들어있는지)

Given an array of integers and a value, determine if there are any two integers in the array whose sum is equal to the given value. Return true if the sum exists and return false if it does not. Consider this array and the target sums:

let findSumOfTwo = function(A, val) {
let foundValues = new Set();
for (let a in A) {
if (foundValues.has(val - A[a])) {
return true;
}

foundValues.add(A[a]);
}
return false;
};

let v = [5, 7, 1, 2, 8, 4, 3];
let test = [3, 20, 1, 2, 7];

}

7. Most Commonly Used Character in String (제일 많이 쓰인 글자 찾기)

Write a function that takes a string, and returns the character that is most commonly used in the string.

const countingChars = (str) => {
const charCount = {}; //Step 0
let maxCharCount = 0;
let maxChar = '';
for(let i = 0; i < str.length; i++) { // Step 1.
charCount[str[i]] = ++charCount[str[i]] || 1;
}


for(let key in charCount) { //Step 2
if(charCount[key] >= maxCharCount) {
maxCharCount = charCount[key] //Step 3
maxChar = key;
}
}
return maxChar; //Step 4
}
